Quick Examples

1. Compute Derivative (Tier 1)

local
    l_calc: SIMPLE_CALCULUS
    l_f: FUNCTION [REAL_64, REAL_64]
    l_result: REAL_64
do
    create l_calc
    l_f := agent (x: REAL_64): REAL_64 do Result := x * x end
    l_result := l_calc.derivative (l_f, 1.0)
    -- l_result is approximately 2.0 (derivative of x^2 at x=1)
end

2. Compute Integral (Tier 1)

local
    l_calc: SIMPLE_CALCULUS
    l_f: FUNCTION [REAL_64, REAL_64]
    l_result: REAL_64
do
    create l_calc
    l_f := agent (x: REAL_64): REAL_64 do Result := x end
    l_result := l_calc.integrate (l_f, 0.0, 1.0)
    -- l_result is approximately 0.5 (integral of x from 0 to 1)
end

3. Compute with Error Bounds (Tier 2)

local
    l_calc: SIMPLE_CALCULUS
    l_f: FUNCTION [REAL_64, REAL_64]
    l_result: DERIVATIVE_RESULT
do
    create l_calc
    l_f := agent (x: REAL_64): REAL_64 do Result := x * x end
    l_result := l_calc.derivative_with_options (l_f, 1.0, 4, 1.0e-6)
    -- l_result.value: computed derivative
    -- l_result.error_bound: estimated error
    -- l_result.convergence_status: CONVERGED/WARNING/FAILED
end

4. Compute Partial Derivative (Tier 1)

local
    l_calc: SIMPLE_CALCULUS
    l_f: FUNCTION [ARRAY [REAL_64], REAL_64]
    l_point: ARRAY [REAL_64]
    l_result: REAL_64
do
    create l_calc
    -- f(x,y) = x^2 + y^2, compute ∂f/∂x at (1,1)
    l_f := agent (v: ARRAY [REAL_64]): REAL_64
        do Result := v[1]*v[1] + v[2]*v[2] end
    create l_point.make_filled (0.0, 1, 2)
    l_point[1] := 1.0
    l_point[2] := 1.0
    l_result := l_calc.partial_derivative (l_f, l_point, 1)
end

5. Compute Gradient (Tier 2)

local
    l_calc: SIMPLE_CALCULUS
    l_f: FUNCTION [ARRAY [REAL_64], REAL_64]
    l_point: ARRAY [REAL_64]
    l_grad: ARRAY [REAL_64]
do
    create l_calc
    l_f := agent (v: ARRAY [REAL_64]): REAL_64
        do Result := v[1]*v[1] + v[2]*v[2] end
    create l_point.make_filled (0.0, 1, 2)
    l_point[1] := 1.0
    l_point[2] := 2.0
    l_grad := l_calc.gradient (l_f, l_point)
    -- l_grad[1] ≈ 2.0 (∂f/∂x = 2x at x=1)
    -- l_grad[2] ≈ 4.0 (∂f/∂y = 2y at y=2)
end

6. Expert API - Direct Engine Access (Tier 3)

local
    l_calc: SIMPLE_CALCULUS
    l_engine: DERIVATIVE_CALCULATOR
    l_f: FUNCTION [REAL_64, REAL_64]
do
    create l_calc
    l_engine := l_calc.create_derivative_calculator
    l_f := agent (x: REAL_64): REAL_64 do Result := x * x end
    -- Access three different methods directly
    Result := l_engine.forward_difference (l_f, 1.0)
    Result := l_engine.backward_difference (l_f, 1.0)
    Result := l_engine.central_difference (l_f, 1.0)
end

Common Operations

API Tiers

  • Tier 1 (Simple) — `derivative`, `integrate`, `partial_derivative`, `gradient`
    Returns single values or arrays
  • Tier 2 (Advanced) — `derivative_with_options`, `integrate_with_options`
    Returns DERIVATIVE_RESULT or INTEGRAL_RESULT with error bounds and status
  • Tier 3 (Expert) — `create_derivative_calculator`, `create_integral_calculator`
    Direct engine access for research and custom algorithms

Methods

Method Purpose Returns
derivative (f, x) Compute df/dx at point x REAL_64
integrate (f, a, b) Compute ∫[a,b] f(x)dx REAL_64
partial_derivative (f, point, i) Compute ∂f/∂x_i at point REAL_64
gradient (f, point) Compute all ∂f/∂x_i ARRAY [REAL_64]
derivative_with_options (f, x, method, tol) Derivative with control DERIVATIVE_RESULT
integrate_with_options (f, a, b, method, tol) Integration with control INTEGRAL_RESULT

Derivative Methods (for derivative_with_options)

  • 1 = Forward difference
  • 2 = Backward difference
  • 3 = Central difference
  • 4 = All three with comparison (recommended)

Integration Methods (for integrate_with_options)

  • 1 = Trapezoidal rule (simple, less accurate)
  • 2 = Simpson's rule (moderate accuracy)
  • 3 = Adaptive Simpson (recommended, highest accuracy)

Result Types

DERIVATIVE_RESULT

-- Features:
-- value: REAL_64 - computed derivative
-- error_bound: REAL_64 - estimated error
-- convergence_status: INTEGER - CONVERGED/WARNING/FAILED
-- is_converged: BOOLEAN - true if CONVERGED
-- is_warning: BOOLEAN - true if WARNING
-- is_failed: BOOLEAN - true if FAILED

INTEGRAL_RESULT

-- Features:
-- value: REAL_64 - computed integral
-- error_bound: REAL_64 - estimated error
-- iterations: INTEGER - number of iterations performed
-- is_accurate: BOOLEAN - true if error_bound < 1e-5
